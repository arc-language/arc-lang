# Arc NVIDIA GPU Execution Model

This document explains how Arc's GPU execution works from end-to-end using the `async func<gpu>` and `await` syntax to interface with NVIDIA's CUDA Driver API.

---

## The Complete Flow

```
Arc Source (async func<gpu> + await)
    ↓
Arc Parser → AST
    ↓
Arc IR Generator → SSA IR
    ↓
PTX Backend → PTX Assembly (text)
    ↓
Arc Runtime (via extern cuda driver API)
    ↓
cuModuleLoadData(ptx_string)
    ↓
NVIDIA Driver JIT → SASS Machine Code
    ↓
GPU Execution → await synchronizes & returns result
```

---

## Step-by-Step Breakdown

### Step 1: User Writes GPU Function with async/await Syntax

```arc
// user_code.arc
async func double_array<gpu>(arr: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    
    let idx = gpu.thread_id()
    if idx < n {
        result[idx] = arr[idx] * 2.0
    }
    
    return result
}

func main() {
    let data = gpu.unified_malloc<float32>(1024)
    
    // Initialize on CPU
    for i in 0..1024 {
        data[i] = cast<float32>(i)
    }
    
    // Execute on GPU - await triggers GPU execution and synchronization
    let result = await double_array(data, 1024)
    
    io.printf("Result: %f\n", result[0])
    
    gpu.unified_free(data)
    gpu.unified_free(result)
}
```

### Step 2: Compiler Detects GPU Execution

When the Arc compiler sees:
```arc
async func double_array<gpu>(arr: *float32, n: usize) *float32 { ... }
```

It recognizes this as:
1. **GPU function** (`<gpu>` marker - compile to PTX)
2. **Async execution** (`async func` - returns immediately, runs in background)
3. **Awaitable** (`await` synchronizes and returns result)

When it sees:
```arc
let result = await double_array(data, 1024)
```

It knows:
1. **Launch on default device** (device 0 when no device specified)
2. **Synchronize** (wait for GPU to complete)
3. **Return result**

The compiler extracts the kernel and compiles it separately:

```
Main Program Path:          GPU Kernel Path:
Arc → IR → AMD64            Arc → IR → PTX
```

### Step 3: PTX Generation

The compiler generates PTX assembly from the kernel function:

```ptx
.version 7.0
.target sm_75
.address_size 64

.visible .entry double_array(
    .param .u64 arr,
    .param .u64 n,
    .param .u64 result
) {
    .reg .u32 %tid;
    .reg .u64 %r<4>;
    .reg .f32 %f<2>;
    .reg .pred %p1;
    
    // let idx = gpu.thread_id()
    mov.u32 %tid, %tid.x;
    
    // if idx < n
    ld.param.u64 %r1, [n];
    cvt.u64.u32 %r2, %tid;
    setp.lt.u64 %p1, %r2, %r1;
    @!%p1 bra EXIT;
    
    // Calculate address: arr + idx * 4
    mul.wide.u32 %r3, %tid, 4;
    ld.param.u64 %r4, [arr];
    add.u64 %r4, %r4, %r3;
    
    // Load arr[idx]
    ld.global.f32 %f1, [%r4];
    
    // Multiply by 2.0
    mul.f32 %f1, %f1, 2.0;
    
    // Calculate result address and store
    ld.param.u64 %r5, [result];
    add.u64 %r5, %r5, %r3;
    st.global.f32 [%r5], %f1;
    
EXIT:
    ret;
}
```

### Step 4: PTX Embedded in Binary

The compiler embeds the PTX string into the Arc executable:

```arc
// Generated by compiler (conceptual)
const __double_array_ptx: string = ".version 7.0\n.target sm_75\n..."
```

Arc interfaces with NVIDIA's CUDA Driver API using `extern`:

## Complete Runtime Implementation (Pure Arc)

```arc
// lib/gpu/runtime.arc
namespace gpu

struct DeviceContext {
    device_id: int32
    device: int32
    context: *void
}

let devices: array<DeviceContext, 8> = []
let device_count: int32 = 0
let initialized: bool = false

func init() {
    if initialized {
        return
    }
    
    // Initialize CUDA
    let result = cuda.cuInit(0)
    if result != 0 {
        raise("CUDA initialization failed")
    }
    
    // Get device count
    result = cuda.cuDeviceGetCount(&device_count)
    if result != 0 {
        raise("Failed to get CUDA device count")
    }
    
    // Initialize all devices
    for i in 0..device_count {
        let mut dev_ctx: DeviceContext
        dev_ctx.device_id = i
        
        result = cuda.cuDeviceGet(&dev_ctx.device, i)
        if result != 0 {
            raise("Failed to get CUDA device {i}")
        }
        
        result = cuda.cuCtxCreate(&dev_ctx.context, 0, dev_ctx.device)
        if result != 0 {
            raise("Failed to create CUDA context for device {i}")
        }
        
        devices[i] = dev_ctx
    }
    
    initialized = true
}

func set_device(device_id: int32) {
    if device_id >= device_count {
        raise("Device {device_id} not available (system has {device_count} GPUs)")
    }
    cuda.cuCtxSetCurrent(devices[device_id].context)
}

func sync() {
    cuda.cuCtxSynchronize()
}

func device_count() int32 {
    init()
    return device_count
}

struct GPUKernel {
    module: *void
    function: *void
    name: string
}

func load_ptx(ptx_code: string, kernel_name: string) GPUKernel {
    init()  // Ensure CUDA is initialized
    
    let module: *void = null
    
    // Load PTX module
    let result = cuda.cuModuleLoadData(&module, ptx_code)
    if result != 0 {
        raise("Failed to load PTX module")
    }
    
    // Get kernel function
    let function: *void = null
    result = cuda.cuModuleGetFunction(&function, module, kernel_name)
    if result != 0 {
        raise("Failed to get kernel function")
    }
    
    return GPUKernel{
        module: module,
        function: function,
        name: kernel_name
    }
}

func launch(kernel: GPUKernel, grid_dim: uint32, block_dim: uint32, args: **void) {
    let result = cuda.cuLaunchKernel(
        kernel.function,
        grid_dim, 1, 1,      // Grid dimensions
        block_dim, 1, 1,     // Block dimensions
        0,                   // Shared memory
        null,                // Stream
        args,                // Kernel arguments
        null                 // Extra
    )
    
    if result != 0 {
        raise("Kernel launch failed")
    }
}

// Unified memory allocation (simplest API)
func unified_malloc<T>(count: usize) *T {
    init()
    let size = count * sizeof<T>
    let ptr: *void = null
    
    let result = cuda.cuMemAllocManaged(&ptr, size, 1)  // 1 = CU_MEM_ATTACH_GLOBAL
    if result != 0 {
        raise("Unified memory allocation failed")
    }
    
    return cast<*T>(ptr)
}

func unified_free(ptr: *void) {
    cuda.cuMemFree(ptr)
}

// Special function for getting thread ID (compiler intrinsic)
func thread_id() usize {
    // This is replaced by the compiler with PTX: mov.u32 %tid, %tid.x
    // In the kernel context only
    return 0  // Placeholder
}
```

## How the Compiler Transforms async func<gpu> + await

### Original Arc Code:
```arc
async func double_values<gpu>(arr: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    let idx = gpu.thread_id()
    if idx < n {
        result[idx] = arr[idx] * 2.0
    }
    return result
}

func main() {
    // Default device (0)
    let result = await double_values(data, 1024)
    
    // Explicit device
    let result2 = await(1) double_values(data, 1024)
}
```

### Compiler Transformation:

**1. Extract kernel function**
```arc
async func double_values<gpu>(arr: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    let idx = gpu.thread_id()
    if idx < n {
        result[idx] = arr[idx] * 2.0
    }
    return result
}
```

**2. Generate PTX from kernel**
```arc
const __double_values_ptx: string = "
.version 7.0
.target sm_75
.address_size 64
.visible .entry double_values(
    .param .u64 arr,
    .param .u64 n,
    .param .u64 result
) {
    // ... PTX assembly ...
    ret;
}
"
```

**3. Transform await call into runtime calls**
```arc
// Original:
let result = await double_values(data, 1024)

// Becomes:
{
    // Default to device 0
    gpu.set_device(0)
    
    // Allocate result buffer
    let result = gpu.unified_malloc<float32>(1024)
    
    // Load kernel
    let kernel = gpu.load_ptx(__double_values_ptx, "double_values")
    
    // Prepare arguments
    let args: array<*void, 3> = {
        cast<*void>(&data),
        cast<*void>(&1024),
        cast<*void>(&result)
    }
    
    // Launch kernel (async)
    gpu.launch(kernel, 4, 256, &args[0])  // 4 blocks × 256 threads = 1024 threads
    
    // await = synchronize before returning
    gpu.sync()
    
    result  // Return the result
}
```

**4. Transform await(device) call**
```arc
// Original:
let device_id = 1
let result = await(device_id) double_values(data, 1024)

// Becomes:
{
    // Use specified device
    gpu.set_device(device_id)
    
    // ... rest is the same as above ...
    
    gpu.sync()
    result
}
```

## Memory Management Examples

### Unified Memory (Simplest - 90% Use Case)

```arc
async func process_array<gpu>(arr: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    let idx = gpu.thread_id()
    if idx < n {
        result[idx] = arr[idx] * 2.0
    }
    return result
}

func example_unified() {
    // Allocate memory accessible from both CPU and GPU
    let data = gpu.unified_malloc<float32>(1024)
    
    // Write from CPU
    for i in 0..1024 {
        data[i] = cast<float32>(i)
    }
    
    // Process on GPU - await synchronizes automatically
    let result = await process_array(data, 1024)
    
    // Read from CPU (result is already synced)
    io.printf("Result: %f\n", result[0])
    
    gpu.unified_free(data)
    gpu.unified_free(result)
}
```

### Explicit Memory Management (Advanced - Performance)

```arc
async func process_gpu_memory<gpu>(gpu_arr: *float32, n: usize, gpu_result: *float32) {
    let idx = gpu.thread_id()
    if idx < n {
        gpu_result[idx] = gpu_arr[idx] * 2.0
    }
}

func example_explicit() {
    // CPU memory
    let cpu_data = alloca(float32, 1024)
    for i in 0..1024 {
        cpu_data[i] = cast<float32>(i)
    }
    
    // GPU memory
    let gpu_data: *float32 = null
    let gpu_result: *float32 = null
    cuda.cuMemAlloc(cast<**void>(&gpu_data), 1024 * sizeof<float32>)
    cuda.cuMemAlloc(cast<**void>(&gpu_result), 1024 * sizeof<float32>)
    
    // Copy CPU → GPU
    cuda.cuMemcpyHtoD(gpu_data, cpu_data, 1024 * sizeof<float32>)
    
    // Execute kernel
    await process_gpu_memory(gpu_data, 1024, gpu_result)
    
    // Copy GPU → CPU
    cuda.cuMemcpyDtoH(cpu_data, gpu_result, 1024 * sizeof<float32>)
    
    // Cleanup
    cuda.cuMemFree(gpu_data)
    cuda.cuMemFree(gpu_result)
    
    io.printf("Result: %f\n", cpu_data[0])
}
```

## Grid and Block Dimension Calculation

The compiler calculates grid/block dimensions automatically:

```arc
// Compiler calculates grid/block dimensions
func calculate_grid_dims(total_threads: usize) (uint32, uint32) {
    const BLOCK_SIZE: uint32 = 256  // Threads per block
    
    let blocks = (total_threads + BLOCK_SIZE - 1) / BLOCK_SIZE
    
    return (cast<uint32>(blocks), BLOCK_SIZE)
}

// Example: 1024 threads
// blocks = (1024 + 255) / 256 = 4
// Grid: 4 blocks × 256 threads/block = 1024 threads
```

For the kernel:
```arc
let result = await process_data(arr, 1024)
```

The compiler knows:
- `n = 1024` (the size parameter)
- Each thread processes one element
- Launch `ceil(1024 / 256) = 4` blocks of 256 threads

## Complete Example: Matrix Multiply

```arc
async func matmul<gpu>(A: *float32, B: *float32, C: *float32, N: usize) {
    let total = N * N
    let idx = gpu.thread_id()
    
    if idx < total {
        let row = idx / N
        let col = idx % N
        
        let mut sum: float32 = 0.0
        for k in 0..N {
            sum += A[row * N + k] * B[k * N + col]
        }
        C[row * N + col] = sum
    }
}

func main() {
    const N: usize = 1024
    
    // Allocate unified memory
    let A = gpu.unified_malloc<float32>(N * N)
    let B = gpu.unified_malloc<float32>(N * N)
    let C = gpu.unified_malloc<float32>(N * N)
    
    // Initialize matrices on CPU
    for i in 0..(N * N) {
        A[i] = 1.0
        B[i] = 2.0
    }
    
    // Compute on GPU - await synchronizes automatically
    await matmul(A, B, C, N)
    
    // Result is in C, already synchronized
    io.printf("C[0,0] = %f\n", C[0])
    
    // Cleanup
    gpu.unified_free(A)
    gpu.unified_free(B)
    gpu.unified_free(C)
}
```

## Composable GPU Pipelines

The async/await syntax enables natural pipeline composition:

```arc
async func preprocess<gpu>(data: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    let idx = gpu.thread_id()
    if idx < n {
        result[idx] = normalize(data[idx])
    }
    return result
}

async func apply_model<gpu>(data: *float32, weights: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    let idx = gpu.thread_id()
    if idx < n {
        result[idx] = relu(data[idx] * weights[idx])
    }
    return result
}

async func postprocess<gpu>(data: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    let idx = gpu.thread_id()
    if idx < n {
        result[idx] = softmax(data[idx])
    }
    return result
}

func inference_pipeline(input: *float32, weights: *float32, n: usize) *float32 {
    // Chain GPU operations - each await synchronizes that stage
    let normalized = await preprocess(input, n)
    let activated = await apply_model(normalized, weights, n)
    let output = await postprocess(activated, n)
    
    gpu.unified_free(normalized)
    gpu.unified_free(activated)
    
    return output
}
```

## Compiler Internals: PTX Code Generation

When the compiler encounters `gpu.thread_id()`:

```arc
// Arc IR
%tid = call @gpu.thread_id()

// PTX Generation
mov.u32 %tid, %tid.x
```

When the compiler encounters array access:

```arc
// Arc code
result[idx] = arr[idx] * 2.0

// Arc IR
%1 = mul %idx, 4              // sizeof(float32) = 4
%2 = add %arr_base, %1        // Calculate address
%3 = load %2                  // Load value
%4 = fmul %3, 2.0             // Multiply
%5 = add %result_base, %1     // Calculate result address
store %4, %5                  // Store result

// PTX
mul.wide.u32 %r1, %tid, 4
add.u64 %r2, %arr_base, %r1
ld.global.f32 %f1, [%r2]
mul.f32 %f1, %f1, 2.0
add.u64 %r3, %result_base, %r1
st.global.f32 [%r3], %f1
```

## Multi-GPU Training Pipeline

```arc
async func train_batch<gpu>(model: *float32, data: *float32, labels: *float32, n: usize) *float32 {
    let gradients = gpu.unified_malloc<float32>(n)
    let idx = gpu.thread_id()
    if idx < n {
        // Forward pass
        let pred = forward(model, data[idx])
        // Backward pass
        gradients[idx] = compute_gradient(pred, labels[idx])
    }
    return gradients
}

func distributed_training(batches: array<*float32, 4>, model: *float32) {
    let num_gpus = gpu.device_count()
    
    if num_gpus < 4 {
        io.printf("Warning: Only {num_gpus} GPUs available, need 4 for optimal performance\n")
    }
    
    // Train 4 batches in parallel across GPUs
    let grad0 = await(0) train_batch(model, batches[0], labels[0], batch_size)
    let grad1 = await(1) train_batch(model, batches[1], labels[1], batch_size)
    let grad2 = await(2) train_batch(model, batches[2], labels[2], batch_size)
    let grad3 = await(3) train_batch(model, batches[3], labels[3], batch_size)
    
    // Average gradients (data parallelism)
    let avg_grad = average_gradients(grad0, grad1, grad2, grad3)
    
    // Update model
    update_weights(model, avg_grad)
    
    // Cleanup
    gpu.unified_free(grad0)
    gpu.unified_free(grad1)
    gpu.unified_free(grad2)
    gpu.unified_free(grad3)
    gpu.unified_free(avg_grad)
}
```

## Key Points

1. **`async func<gpu>` marks GPU compilation** - Compiler generates PTX for these functions
2. **`await` uses default device** - Simple, works 90% of the time (device 0)
3. **`await(N)` for explicit device** - Runtime device selection when needed
4. **PTX is embedded** - Compiler embeds PTX strings in the Arc binary
5. **JIT happens at runtime** - First kernel launch compiles PTX → SASS (cached)
6. **Unified memory is easiest** - CPU and GPU share the same pointers
7. **Synchronization is explicit** - `await` blocks until GPU completes
8. **Composable pipelines** - Functions return values, enabling chaining
9. **No separate runtime needed** - Arc uses `extern` to call CUDA Driver API directly
10. **Multi-GPU is natural** - Use `await(device_id)` with runtime variables

## The Magic Moment

When you write:
```arc
async func double_values<gpu>(data: *float32, n: usize) *float32 {
    let result = gpu.unified_malloc<float32>(n)
    result[gpu.thread_id()] = data[gpu.thread_id()] * 2.0
    return result
}

let result = await double_values(array, 1024)
```

**What actually happens:**

1. Compiler sees `async func<gpu>` → generates PTX assembly
2. PTX embedded in binary as string constant
3. At runtime when `await` is encountered:
   - Switches to device 0 (default)
   - `cuModuleLoadData(ptx_string)` loads to GPU
   - NVIDIA driver JIT compiles PTX → SASS (once, then cached)
   - `cuLaunchKernel()` executes 1024 threads in parallel
   - Each thread runs with a different `thread_id()`
   - `await` blocks and calls `cuCtxSynchronize()` 
   - Returns result when GPU completes
4. Result is synchronized and ready to use

**All with clean, intuitive async/await syntax.** 🚀

```arc
    let data = gpu.unified_malloc<float32>(1000)
    // ... initialize data ...
    
    // Simple - just await (uses device 0)
    let result = await process(data, 1000)
    
    io.printf("Done: %f\n", result[0])
}
```

### Intermediate - Explicit Device Control

```arc
func main() {
```

### Advanced - Dynamic Multi-GPU

```arc
func main() {
    let data = load_huge_dataset()
    let num_gpus = gpu.device_count()
    let chunk_size = data.len / num_gpus
    
    let results: array<*float32, 8>
    
    // Dynamically distribute across all GPUs
    for i in 0..num_gpus {
        let chunk_start = i * chunk_size
        results[i] = await(i) process(&data[chunk_start], chunk_size)
    }
    
    // Combine results
    let final = combine(results, num_gpus)
    io.printf("Processed on {num_gpus} GPUs\n")
}
```

## Comparison: CPU async vs GPU async

```arc
// CPU async (I/O, network, etc.)
async func fetch_url(url: string) *byte {
    // Network I/O
}

async func read_file(path: string) *byte {
    // File I/O
}

// GPU async (parallel computation)
async func compute_intensive<gpu>(data: *float32, n: usize) *float32 {
    // Parallel GPU computation
}

async func train_neural_net<gpu>(data: *float32, epochs: int) *float32 {
    // GPU training
}

func main() {
    // CPU async - no device specifier
    let web_data = await fetch_url("https://api.com/data")
    let file_data = await read_file("config.json")
    
    // GPU async - default device (90% use case)
    let processed = await compute_intensive(web_data, 10000)
    
    // GPU async - explicit device (10% use case)
    let device = 1
    let trained = await(device) train_neural_net(processed, 100)
}
```

The `<gpu>` marker in `async func<gpu>` tells the compiler this is GPU code (compile to PTX), while the absence of `<gpu>` means regular CPU async. The `await` vs `await(device)` at the call site determines device selection. Clean, simple, powerful! ✨
